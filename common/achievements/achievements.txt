How_Did_We_Get_Here = {
	
	possible = {
		difficulty > 1
		is_major = no
	}
	
	happened = {
		hidden_trigger = {
			set_temp_variable = { num_of_major_enemies = 0 }
			all_enemy_country = {
				if = {
					limit = {
						is_major = yes
					}
					add_to_temp_variable = { num_of_major_enemies = 1 }
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = How_Did_We_Get_Here_tt
			check_variable = { num_of_major_enemies > 4 }
		}
	}
}
