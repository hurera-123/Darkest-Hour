# Generic Nuclear Attack on Majors
news_event = {
	id = Nuke_Other.0
	title = Nuke_Other.0.t
	desc = Nuke_Other.0.d
	picture = GFX_news_event_nuke

	major = yes

	is_triggered_only = yes

	trigger = {
		NOT = {
			FROM = {
				FROM = {
					OR = {
						state = 528
						state = 529
						state = 282
						state = 219
						state = 195
						state = 1057
						state = 809
						state = 2
						state = 160
						state = 16
						state = 126
						state = 341
						state = 47
						state = 608
						state = 784
						state = 358
						state = 378
						state = 285
						state = 4
						state = 7
					}
				}
			}
		}
	}

	option = {
		name = Nuke_Other.0.A
		trigger = {
			NOT = { owns_state = FROM.FROM }
			NOT = { any_enemy_country = { controls_state = FROM.FROM } }
		}
	}
	
	option = {
		name = Nuke_Other.0.B
		trigger = {
			owns_state = FROM.FROM
		}
	}
	
	option = {
		name = Nuke_Other.0.C
		trigger = {
			any_enemy_country = {
				controls_state = FROM.FROM
			}
		}
	}
}
